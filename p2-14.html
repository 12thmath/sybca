Programe - 14.write a programe using pointer to read an array of n elements and print all the elemets of array in reverse order.

void main()
{
int i,n,a[100];
int *p;

printf("enter the number of elements");
scanf("%d",&n);

for(i=0;i {
printf("enter a[%d]::",i);
scanf("%d",&a[i]);
}
p=&a[n-1];

for(i=n-1;i>=0;i--)
{
printf("\n a[%d]=%d",i,*p);
p--;
}
}
