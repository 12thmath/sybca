Programe - 20.write a programe to find total numbers of uppercase letters,lowercase letters, blank space and special charater from the given file.

void main()
{
char fname[100],ch;
int c=0,s=0,d=0,b=0,sp=0;
FILE *f;

printf("enter file name==");
gets(fname);

fopen=(fname,"r");

if(f==null)
printf("file is not present");
else
{
printf("\n file exist");
printf("\n file content.....");
printf("\n\n");
while((ch=getc(f))!=EOF)
{
printf("%c",ch);
if(ch!='\n')
{
if(ch>=65&&ch<=90)
c++;
else if(ch>='a'&&ch<='z')
s++;
else if(ch>='0'&&<='9')
d++;
else if(ch==32)
b++;
else
sp++;
}
}
}

printf("\n total no of uppercase character=%d",c);
printf("\n total no of lowercase character=%d",s);
printf("\n total no of digit=%d",d);
printf("\n total no of blank space=%d",b);
printf("\n total no of special character=%d",sp);

fclose(f);
}
