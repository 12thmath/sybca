Programe - 11.write a c programe to read student information like rollnumber, name and marks of 3 subjects using structure.
pass the structure as an argument to udf function and print the data in function.

void disp(struct stud);
struct stud
{
int no;
char name[20];
int m1,m2,m3;
}s;

void main()
{
printf("enter the student no");
scanf("%d",&s.no);
printf("enter student name");
gets(s.name);
printf("enter mark of subject1");
scanf("%d",&s.m1);
printf("enter mark of subject2");
scanf("%d",&s.m2);
printf("enter mark of subject3");
scanf("%d",&s.m3);

disp (s);
}

void disp(struct stud s)
{
printf("\n sno=%d",s.no);
printf("\n sname=%s",s.name);
printf("\n marks of subject1%d",s.m1);
printf("\n marks of subject2%d",s.m2);
printf("\n marks of subject3%d",s.m3);
}
