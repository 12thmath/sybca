Programe - 9.write a programe to read employee no, emploee code, emploee name, basic salary and lic.
calculate da, hra, ma, pf, it and pt according to following criteria:-

void main()
{
int eno,code;
char name[20];
float basic,da,hra,ma,pf,lic,pt,totded,gsal,netsal,it;

printf("\n enter employee number:");
scanf("%d",&eno);
printf("\n enter employee code:");
scanf("%d",&code);
printf("\n enter employee name:");
gets(name);
printf("\n enter basic salary:");
scanf("%f",&basic);
printf("\n enter lic amount:");
scanf("%f",&lic);

//da and pf calculation
if(basic<=5000)
{
da=basic*0.20;
pf=basic*0.08;
}
else if(basic<=9000)
{
da=basic*0.30;
pf=basic*0.12;
}
else if(basic<=15000)
{
da=basic*0.40;
pf=basic*0.20;
}
else
{
da=basic*0.50;
pf=basic*0.25;
}

//hra calculation
switch(code)
{
case 1:
hra=120;
break;
case 2:
hra=250;
break;
case 3:
hra=400;
break;
default:
hra=0;
}

//ma calculation
ma=basic*0.20;
gsal=basic+da+hra+ma;

//pt calculation
if(gsal<=10000)
pt=200;
else if(gsal<=20000)
pt=400;
else
pt=500;

//it calculation
it=gsal*0.10;
totded=pf+it+pt+lic;
netsal=gsal-totded;

if(code>=0&&code<=3)
{
printf("\n\t\t employee payslip");
printf("\n=================================================================");
printf("\n employee code=%d\t\temployee name:%s",code,name);
printf("\n=================================================================");
printf("\n earings\tamount\t\tdeduction\tamount");
printf("\n=================================================================");
printf("\n basic\t\t%7.2f\t\tpt\t\t%7.2f",basic,pf);
printf("\n da\t\t%7.2f\t\tlic\t\t%7.2f",da,lic);
printf("\n hra\t\t%7.2f\t\tit\t\t%7.2f",hra,it);
printf("\n ma\t\t%7.2f\t\tpt\t\t%7.2f",ma,pt);
printf("\n=================================================================");
printf("\n gross\t\t%7.2f\t\ttotal deduction\t%7.2f",gsal,totded);
printf("\n\t\t\t\t\t net amount%7.2f",netsal);
}
else
{
printf("invalid code");
}
}
